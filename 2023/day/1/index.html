get_last_number <- function(x) {
  found_reversed_string <-
    stringi::stri_reverse(x) %>%
    str_extract(regex_reversed)

  digits_reversed[found_reversed_string] %>%
    set_names(NULL)
}

get_first_number <- function(x) {
  found_string <- str_extract(
    x,
